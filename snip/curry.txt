-- Haskell
ghc> let add x y = x + y    -- add :: (Num a) => a -> a -> a
ghc> add 2 3
5
ghc> let addTwo = add 2
ghc> addTwo 3
5

----------

// JS
var add = function(x, y) { return x + y; };
add(2, 3); // -> 5

var addTwo = (function(x) {
    return function(y) { return add(x, y); };
})(2);
addTwo(3); // -> 5
